  public static String convertAggregateBoxToJsonString(AggregateBox aggregateBox) {
    try {
      return OBJECT_MAPPER.writeValueAsString(aggregateBox);
    } catch (JsonProcessingException ex) {
      // TODO Handle.
      return null;
    }
  }

  public static AggregateBox convertJsonStringToAggregateBox(String json) {
    try {
      return OBJECT_MAPPER.readValue(json, AggregateBox.class);
    } catch (Exception ex) {
      return null;
    }
  }

    final KStream<String, UC1DeviceSpecificEvent> stringUC1DeviceSpecificEventKStream =
        streamsBuilder.stream(
            UC1Constants.SENSOR_TELEMETRY_EVENT_TOPIC,
            Consumed.with(STRING_SERDE, UC_1_DEVICE_SPECIFIC_EVENT_SERDE));

    KStream<String, UC1DeviceGenericEvent> deviceGenericEventKStream =
        stringUC1DeviceSpecificEventKStream

                  UC1DeviceGenericEvent temperature =

                  UC1DeviceGenericEvent moisture =
                      uc1DeviceGenericEventFactory.createMoisture(value);
                  return Arrays.asList(temperature, moisture);
                })
            .map(
                new KeyValueMapper<
                    String, UC1DeviceGenericEvent, KeyValue<String, UC1DeviceGenericEvent>>() {

                  @Override
                  public KeyValue<String, UC1DeviceGenericEvent> apply(
                      String ignored, UC1DeviceGenericEvent uc1GenericEvent) {
                    return new KeyValue<>(uc1GenericEvent.getDeviceId(), uc1GenericEvent);
                  }
                });
    //        .groupBy(
    //            (key, value) -> String.format("%s:%s", value.getKey(), value.getDeviceId()),
    //            Grouped.with(STRING_SERDE, UC_1_DEVICE_GENERIC_EVENT_SERDE))

    //        .reduce((key, value) -> value)

    KStream<String, UC1DeviceGenericEvent> temperatureEventKStream =
        deviceGenericEventKStream.filter((key, value) -> "TEMPERATURE".equals(value.getKey()));

    KStream<String, UC1DeviceGenericEvent> moistureEventKStream =
        deviceGenericEventKStream.filter((key, value) -> "MOISTURE".equals(value.getKey()));

    KeyValueBytesStoreSupplier storeSupplier = Stores.inMemoryKeyValueStore("queryable-store-name");

    //    streamsBuilder.addStateStore(

    temperatureEventKStream
        .map(
            new KeyValueMapper<
                String, UC1DeviceGenericEvent, KeyValue<String, UC1DeviceGenericEvent>>() {

              @Override
              public KeyValue<String, UC1DeviceGenericEvent> apply(
                  String ignored, UC1DeviceGenericEvent uc1GenericEvent) {
                return new KeyValue<>(uc1GenericEvent.getDeviceId(), uc1GenericEvent);
              }
            })
        // .groupByKey(Grouped.with(STRING_SERDE, UC_1_DEVICE_GENERIC_EVENT_SERDE))
      // Not good but not bad either.
        .groupBy(
            (key, value) -> "TEMPERATURE",
            Grouped.with(STRING_SERDE, UC_1_DEVICE_GENERIC_EVENT_SERDE))
        .windowedBy(TimeWindows.of(Duration.ofSeconds(10)).advanceBy(Duration.ofSeconds(1)))
        .aggregate(
            new Initializer<String>() {
              @Override
              public String apply() {
                return convertAggregateBoxToJsonString(new AggregateBox());
              }
            },
            new Aggregator<String, UC1DeviceGenericEvent, String>() {
              @Override
              public String apply(String s, UC1DeviceGenericEvent event, String json) {
                AggregateBox aggregateBox = convertJsonStringToAggregateBox(json);
                aggregateBox.data.put(String.format("device_value_%s", event.getDeviceId()), event.getValue());
                aggregateBox.calculateMean();
                return convertAggregateBoxToJsonString(aggregateBox);
              }
            },
            Materialized.<String, String, WindowStore<Bytes, byte[]>>as(
                    "time-windowed-aggregated-temperature-store2")
                .withCachingDisabled()
                .withKeySerde(STRING_SERDE)
                .withValueSerde(STRING_SERDE))
        .suppress(
            Suppressed.untilTimeLimit(Duration.ofSeconds(10), Suppressed.BufferConfig.unbounded()))
        .toStream()
        .print(Printed.toSysOut());

    temperatureEventKStream
        .map(
            new KeyValueMapper<String, UC1DeviceGenericEvent, KeyValue<String, Double>>() {

              @Override
              public KeyValue<String, Double> apply(
                  String ignored, UC1DeviceGenericEvent uc1GenericEvent) {
                return new KeyValue<>(uc1GenericEvent.getDeviceId(), uc1GenericEvent.getValue());
              }
            })
        // .groupByKey(Grouped.with(STRING_SERDE, DOUBLE_SERDE))
        // TODO Not right but not wrong either
        .groupBy((key, value) -> "TEMPERATURE", Grouped.with(STRING_SERDE, DOUBLE_SERDE))
        .windowedBy(TimeWindows.of(Duration.ofSeconds(10)).advanceBy(Duration.ofSeconds(1)))
        .reduce((key, value) -> value)
        .suppress(
            Suppressed.untilTimeLimit(Duration.ofSeconds(10), Suppressed.BufferConfig.unbounded()))
        //        .aggregate(
        //            new Initializer<String>() {
        //              @Override
        //              public String apply() {

        //              }

  // TODO Move out of this class and rename to UC1GenericAggregateBox
  public static final class AggregateBox implements Serializable {

    private static final long serialVersionUID = 1L;

    public Map<String, Double> data;
    public Double mean;
    public Long timestamp;

    public AggregateBox() {
      this.data = new HashMap<>();
      this.mean = 0D;
      this.timestamp = 0L;
    }

    public void calculateMean() {
      this.mean = 5D;
    }

    @JsonIgnore
    public Instant timestampToInstant() {
      return Instant.ofEpochMilli(this.timestamp);
    }

    @Override
    public String toString() {
      final StringBuffer sb = new StringBuffer("AggregateBox{");
      sb.append("data=").append(data);
      sb.append(", mean=").append(mean);
      sb.append('}');
      return sb.toString();
    }
  }
